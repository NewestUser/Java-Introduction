    void shiftDown() {
        if (currentGear == 0) {

    void speedDown() {
        int nextSpeed = currentSpeed - 10;

        if (nextSpeed >= 0) {

            shiftDown(); // —Å–º—ä–∫–Ω–∏ –ø—Ä–µ–¥–∞–≤–∫–∞

        currentSpeed = 0; // –∑–∞–Ω—É–ª–∏ –æ—Å—Ç–∞—Ç—ä—á–Ω–∞—Ç–∞ —Å–∫–æ—Ä–æ—Å—Ç
    }
